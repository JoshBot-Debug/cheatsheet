# OpenGL

## The graphics pipeline

### 1. **VTX**: Vertex Data
  * The raw data stream fed into the pipeline from the CPU. This includes vertex positions, normals, texture coordinates (UVs), and colors stored in **Vertex Buffer Objects** (**VBOs**).

```cpp
std::vector<GLfloat> vertices = {0.5f,  0.5f, 0.0f, 0.5f, -0.5f, 0.0f, -0.5f, -0.5f, 0.0f, -0.5f,  0.5f, 0.0f};
std::vector<GLuint> indices = {  0, 1, 3, 1, 2, 3};

GLuint VAO, VBO, EBO;
glGenVertexArrays(1, &VAO);
glGenBuffers(1, &VBO);
glGenBuffers(1, &EBO);

// Start recording the VBO layout and EBO binding
glBindVertexArray(VAO);

// Vertex Buffer
glBindBuffer(GL_ARRAY_BUFFER, VBO);
glBufferData(GL_ARRAY_BUFFER, vertices.size() * sizeof(GLfloat), vertices.data(), GL_STATIC_DRAW);

// Element Buffer
glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(GLuint), indices.data(), GL_STATIC_DRAW);

// OpenGL should read the vertex array attributes
// index -> location = 0
// size -> 3 components
// type-> float component
// normalized -> nope
// stride -> Byte offset between consecutive generic vertex attributes. In this case: x,y,z * sizeof(float)
// pointer-> Offset of the first component of the first generic vertex attribute in the array, in this case offset 0.
// https://registry.khronos.org/OpenGL-Refpages/gl4/html/glVertexAttribPointer.xhtml
// void glVertexAttribPointer(GLuint index, GLint size, GLenum type, GLboolean normalized, GLsizei stride, const void * pointer);
glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(float), (void*)0);
// To use non-array data, call glDisableVertexAttribArray.
// To use array data, call glEnableVertexAttribArray. In this case, it's array data.
glEnableVertexAttribArray(0);

// Stop recording
glBindVertexArray(0);
glBindBuffer(GL_ARRAY_BUFFER, 0);

int render() {
  // ...
  glBindVertexArray(VAO);
  glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_INT, 0);
  glBindVertexArray(0);
  // ...
}
```

### 2. **VS**: Vertex Shader
  * Executes once for each vertex.
  * Its primary job is to transform vertices from local object space into Clip Space using your Matrix transformations (MVP). It can also pass data (like texture coordinates) down the line.

```glsl
#version 450 core
layout (location = 0) in vec3 aPos; // Input from VTX
uniform mat4 u_MVP;

void main() {
  gl_Position = u_MVP * vec4(aPos, 1.0); 
}
```

### 3. **TCS**: Tessellation Control Shader (Optional)
  * The first part of the tessellation process. It determines how much to subdivide a patch of geometry by setting tessellation levels (`gl_TessLevelOuter` and `gl_TessLevelInner`).

```glsl
#version 450 core
layout (vertices = 3) out; // Output 3 vertices per patch

void main() {
  // Pass coordinate data through
  gl_out[gl_InvocationID].gl_Position = gl_in[gl_InvocationID].gl_Position;
  
  // Set tessellation levels (how fine the split is)
  if (gl_InvocationID == 0) {
    gl_TessLevelInner[0] = 4.0; // Subdivisions inside
    gl_TessLevelOuter[0] = 4.0; // Subdivisions on edges
    gl_TessLevelOuter[1] = 4.0;
    gl_TessLevelOuter[2] = 4.0;
  }
}
```

### 4. **TES**: Tessellation Evaluation Shader
* Calculates the actual positions of the new vertices generated by the tessellation engine using barycentric coordinates (`gl_TessCoord`).

```glsl
#version 450 core
layout (triangles, equal_spacing, ccw) in;

void main() {
  // Interpolate the position of the new vertex across the triangle patch
  //   glPatchParameteri(GL_PATCH_VERTICES, 3);
  // therefore, gl_in has 3 vertices
  vec3 p0 = gl_in[0].gl_Position.xyz;
  vec3 p1 = gl_in[1].gl_Position.xyz;
  vec3 p2 = gl_in[2].gl_Position.xyz;
  
  // gl_TessCoord holds barycentric coordinates (u, v, w)
  vec3 finalPos = gl_TessCoord.x * p0 + gl_TessCoord.y * p1 + gl_TessCoord.z * p2;
  
  gl_Position = vec4(finalPos, 1.0);
}
```
To render patches via C++, change your draw call from `GL_TRIANGLES` to `GL_PATCHES`:

```cpp
void render() {
  // ...
  glPatchParameteri(GL_PATCH_VERTICES, 3);
  glDrawElements(GL_PATCHES, 6, GL_UNSIGNED_INT, 0);
  // ...
}
```

::::note
`glPatchParameteri(GL_PATCH_VERTICES, 3);` `GL_PATCH_VERTICES` is set to `3`, so in the **TES**, `gl_in` will have 3 vertices/elements
::::

::::info
**Barycentric coordinates** are a system that specifies the location of a point inside or around a geometric shape (a simplex, such as a triangle or tetrahedron) using relative weights or masses assigned to the shape's vertices
::::

### 5. **GS**: Geometry Shader

* Takes full primitives and can generate new vertices or primitives using `EmitVertex()` and `EndPrimitive()`.

```cpp
#version 450 core
layout (triangles) in;
layout (triangle_strip, max_vertices = 3) out;

void main() {
  // Example: Simply pass the original triangle straight through
  for(int i = 0; i < 3; i++) {
    gl_Position = gl_in[i].gl_Position;
    EmitVertex(); // Construct a vertex
  }
  EndPrimitive();
}
```

### 6. **Rasterizer**

```cpp
glViewport(0, 0, 800, 600);          // mapping area to screen space
glCullFace(GL_BACK);                 // Cull back-facing triangles
glEnable(GL_CULL_FACE);              // Enable face culling 
glPolygonMode(GL_FRONT_AND_BACK, GL_FILL); // GL_FILL / GL_LINE / GL_POINT
```

### 7. **FS**: Fragment Shader
* Determines color of the pixel
* Runs once per pixel.

```cpp
#version 450 core
out vec4 FragColor;

void main() {
  FragColor = vec4(1.0f, 0.0f, 0.0f, 1.0f);  // A nice bright red screen :P
}
```

### 8. **FB**: Framebuffer / Per-Sample Operations

* Like the Rasterizer, this is a fixed-function blending and testing ground handled via API commands.

```cpp
glEnable(GL_DEPTH_TEST);  // Discard fragments hidden behind closer objects
glDepthFunc(GL_LESS);

glEnable(GL_BLEND);       // Alpha maps for transparency
glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);

// Clear buffers at start of loop
glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT | GL_STENCIL_BUFFER_BIT);
```

### 9. **CS**: Compute Shader
* Bind textures, write to them, paint a picture, do math, etc. You can use shaders to do any computation.
* **The Compute Shader sits completely outside the traditional graphics rendering pipeline**
```cpp
#version 450 core
layout (local_size_x = 16, local_size_y = 16, local_size_z = 1) in;

layout(rgba32f, binding = 0) writeonly uniform image2D imgOutput;

void main() {
  ivec2 pixelCoords = ivec2(gl_GlobalInvocationID.xy);
  vec4 pixelColor = vec4(1.0, 0.0, 0.0, 1.0);
  
  imageStore(imgOutput, pixelCoords, pixelColor);
}
```

## Fundamental Object Types

Object Name | Acronym | Purpose
|:---|:---|:---
Vertex Array Object | `VAO` | Stores the links between `VBO`/`EBO` bindings and Shader attributes.
Vertex Buffer Object | `VBO` | GPU memory block containing raw array data (Positions, Normals, Colors).
Element Buffer Object | `EBO` | Stores indices determining the sequence in which vertices form triangles.
Frame Buffer Object | `FBO` | An off-screen buffer to render into (useful for post-processing effects).
Render Buffer Object | `RBO` | An optimized write-only canvas (primarily used for offscreen Depth/Stencil testing).

## Standard Matrix Transformations (MVP)

To transform 3D space onto your 2D screen, vertices pass through three core max_vertices
$$
Vertex_{clip} = P \times V \times M \times Vertex_{local}
$$

* **Model Matrix ($M$)**: Moves vertices from Local Object space to World Space (Translation, Rotation, Scale).
* **View Matrix ($V$)**: Moves the world into camera perspective. Defines the positions of Camera Space.
* **Projection Matrix ($P$)**: Projects 3D vectors to 2D normalized screen dimensions (`-1.0` to `1.0`).
  * **Perspective**: Matrix for 3D realism where objects look smaller farther away (`glm::perspective`).
  * **Orthographic**: Matrix for 2D planes/UI components where distance doesn't affect scale (`glm::ortho`).

## Common API Calls

### Context State Configuration
```cpp
glEnable(GL_DEPTH_TEST);  // Enable depth testing (prevents distant fragments overlapping close ones)
glEnable(GL_BLEND);       // Enable alpha blending (Transparency)
glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA); // Default blending function
glClearColor(0.1f, 0.2f, 0.3f, 1.0f);             // Define background clear color
glViewport(0, 0, width, height);                   // Map normalized device coordinates to screen pixels
```

### Buffer Lifecycles (Generate $\rightarrow$ Bind $\rightarrow$ Allocate $\rightarrow$ Delete)
```cpp
GLuint VBO;
glGenBuffers(1, &VBO);
glBindBuffer(GL_ARRAY_BUFFER, VBO);
glBufferData(GL_ARRAY_BUFFER, sizeof(data), data, GL_STATIC_DRAW); // GL_STATIC_DRAW, GL_DYNAMIC_DRAW, GL_STREAM_DRAW
glDeleteBuffers(1, &VBO);
```

::::info
`GL_STATIC_DRAW`: I won't change the buffer data

`GL_DYNAMIC_DRAW`: I may change the buffer data it once in a while

`GL_STREAM_DRAW`: I will change the buffer data it every frame
::::

## Boilerplate

### Compiling a shader program
```cpp
#include <iostream>
#include <string>
#include <vector>
#include <GL/glew.h>

/**
 * @brief Compiles a single shader stage from source code string.
 * @param shaderType GL_VERTEX_SHADER | GL_FRAGMENT_SHADER | GL_GEOMETRY_SHADER | GL_TESS_CONTROL_SHADER | GL_TESS_EVALUATION_SHADER | GL_COMPUTE_SHADER
 * @param shaderSourceCode The raw GLSL source code string.
 * @return GLuint The compiled shader ID, or 0 if compilation failed.
 */
GLuint compileShaderStage(GLenum shaderType, const std::string& shaderSourceCode) {
  std::string stageName = "UNKNOWN";
  switch (shaderType) {
    case GL_VERTEX_SHADER:          stageName = "VERTEX"; break;
    case GL_TESS_CONTROL_SHADER:    stageName = "TESS_CONTROL"; break;
    case GL_TESS_EVALUATION_SHADER: stageName = "TESS_EVALUATION"; break;
    case GL_GEOMETRY_SHADER:        stageName = "GEOMETRY"; break;
    case GL_FRAGMENT_SHADER:        stageName = "FRAGMENT"; break;
    case GL_COMPUTE_SHADER:         stageName = "COMPUTE"; break;
  }

  // Create and source the shader
  GLuint shader = glCreateShader(shaderType);
  const char* sourceCStr = shaderSourceCode.c_str();
  glShaderSource(shader, 1, &sourceCStr, nullptr);
  
  // Compile
  glCompileShader(shader);

  // Check for compilation errors
  GLint success;
  glGetShaderiv(shader, GL_COMPILE_STATUS, &success);
  if (!success) {
    GLint logLength = 0;
    glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &logLength);
    
    std::vector<char> infoLog(logLength);
    glGetShaderInfoLog(shader, logLength, nullptr, infoLog.data());
    
    std::cerr << "ERROR::SHADER::" << stageName << "::COMPILATION_FAILED\n" 
              << infoLog.data() << "\n-----------------------------------------------------\n" 
              << std::endl;
              
    glDeleteShader(shader);
    return 0;
  }

  return shader;
}

/**
 * @brief Links multiple shader stages into a unified shader program.
 * @param shaderStages A vector containing compiled shader handles
 * @return GLuint The linked shader program ID, or 0 if linking failed.
 */
GLuint linkShaderProgram(const std::vector<GLuint>& shaderStages) {
  GLuint program = glCreateProgram();
  
  for (GLuint stage : shaderStages)
    if (stage != 0)
      glAttachShader(program, stage);
  
  glLinkProgram(program);
  
  GLint success;
  glGetProgramiv(program, GL_LINK_STATUS, &success);
  if (!success) {
    GLint logLength = 0;
    glGetProgramiv(program, GL_INFO_LOG_LENGTH, &logLength);
    
    if (logLength > 0) {
        std::vector<char> infoLog(logLength);
        glGetProgramInfoLog(program, logLength, nullptr, infoLog.data());
        std::cerr << "ERROR::SHADER::PROGRAM::LINKING_FAILED\n" 
                  << infoLog.data() << "\n-----------------------------------------------------\n" 
                  << std::endl;
    } else {
        std::cerr << "ERROR::SHADER::PROGRAM::LINKING_FAILED (Unknown Error)\n" << std::endl;
    }
    
    glDeleteProgram(program);
    return 0;
  }
  
  return program;
}

int main() {
  GLuint vs  = compileShaderStage(GL_VERTEX_SHADER,          vsSource);
  GLuint tcs = compileShaderStage(GL_TESS_CONTROL_SHADER,    tcsSource);
  GLuint tes = compileShaderStage(GL_TESS_EVALUATION_SHADER, tesSource);
  GLuint gs  = compileShaderStage(GL_GEOMETRY_SHADER,        gsSource);
  GLuint fs  = compileShaderStage(GL_FRAGMENT_SHADER,        fsSource);
  GLuint cs  = compileShaderStage(GL_COMPUTE_SHADER,         csSource);

  GLuint program = linkShaderProgram({ vs, tcs, tes, gs, fs, cs }); // FYI, this is just an example, cs shaders are generally in a seperate pipeline.

  // Gotta free memory, don't forget
  glDeleteShader(vs);
  glDeleteShader(tcs);
  glDeleteShader(tes);
  glDeleteShader(gs);
  glDeleteShader(fs);
  glDeleteShader(cs);

  // eventually
  glUseProgram(program);
}
```

### Allocating and Binding Mesh Data (`VAO`/`VBO`/`EBO`)

```cpp
std::vector<GLfloat> vertices = {0.5f,  0.5f, 0.0f, 0.5f, -0.5f, 0.0f, -0.5f, -0.5f, 0.0f, -0.5f,  0.5f, 0.0f};
std::vector<GLuint> indices = {  0, 1, 3, 1, 2, 3};

GLuint VAO, VBO, EBO;
glGenVertexArrays(1, &VAO);
glGenBuffers(1, &VBO);
glGenBuffers(1, &EBO);

glBindVertexArray(VAO);

glBindBuffer(GL_ARRAY_BUFFER, VBO);
glBufferData(GL_ARRAY_BUFFER, vertices.size() * sizeof(GLfloat), vertices.data(), GL_STATIC_DRAW);

glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(GLuint), indices.data(), GL_STATIC_DRAW);


// NOTE: Stride is 5 * sizeof(GLfloat) | [x,y,x,u,v]
// Location = 0
// First 3 floats are x,y,z
glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 5 * sizeof(GLfloat), (void*)0);
glEnableVertexAttribArray(0);

// Location = 1
// Following 2 floats are texture coords (UV)
glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 5 * sizeof(GLfloat), (void*)(3 * sizeof(GLfloat)));
glEnableVertexAttribArray(1);

// Unbind VAO & VBO
// DO NOT UNBIND THE EBO, nothing will render :P
// You can unbind the EBO after ending the recording though. But best practice is to never unbind the EBO, doing that is redundant, simply modifies OpenGL state.
glBindVertexArray(0);
glBindBuffer(GL_ARRAY_BUFFER, 0); 
```

::::info
**Multiple `VBO` one `VAO`** & **Multiple `EBO` one `VAO`**

**Multiple `VBO` one `VAO`**:
  * You can use multiple `VBOs` while recording a `VAO`. This is actually a nice practice.
  * *Location = 0* `VBO_postions` for vertex data
  * *Location = 1* `VBO_textures` for texture info
  * *Location = 2* `VBO_normals` for normals

**Multiple `EBO` one `VAO`**:
  * This is **NOT POSSIBLE**
  * You can only bind one `EBO` while recording a `VAO`

**Example**:
```cpp
// Separate raw data streams
std::vector<GLfloat> positions = {};
std::vector<GLfloat> texCoords = {};
std::vector<GLfloat> normals = {};
std::vector<GLuint> indices = {};

// Allocate the Object Handles
GLuint VAO;
GLuint VBO_Positions, VBO_TexCoords, VBO_Normals;
GLuint EBO;

glGenVertexArrays(1, &VAO);
glGenBuffers(1, &VBO_Positions);
glGenBuffers(1, &VBO_TexCoords);
glGenBuffers(1, &VBO_Normals);
glGenBuffers(1, &EBO);

// Start recording
glBindVertexArray(VAO);

// [VBO 1] Setup Positions & link to layout location = 0
glBindBuffer(GL_ARRAY_BUFFER, VBO_Positions);
glBufferData(GL_ARRAY_BUFFER, positions.size() * sizeof(GLfloat), positions.data(), GL_STATIC_DRAW);
glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(GLfloat), (void*)0);
glEnableVertexAttribArray(0);

// [VBO 2] Setup Textures & link to layout location = 1
glBindBuffer(GL_ARRAY_BUFFER, VBO_TexCoords);
glBufferData(GL_ARRAY_BUFFER, texCoords.size() * sizeof(GLfloat), texCoords.data(), GL_STATIC_DRAW);
glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 2 * sizeof(GLfloat), (void*)0); // Size is 2 for U,V!
glEnableVertexAttribArray(1);

// [VBO 3] Setup Normals & link to layout location = 2
glBindBuffer(GL_ARRAY_BUFFER, VBO_Normals);
glBufferData(GL_ARRAY_BUFFER, normals.size() * sizeof(GLfloat), normals.data(), GL_STATIC_DRAW);
glVertexAttribPointer(2, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(GLfloat), (void*)0);
glEnableVertexAttribArray(2);

// [EBO] Single shared index buffer
glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(GLuint), indices.data(), GL_STATIC_DRAW);

// Stop recording
glBindVertexArray(0);
glBindBuffer(GL_ARRAY_BUFFER, 0);
```
::::

### Texturing Pipeline

```cpp
#include <stb_image.h>

GLuint textureID;
glGenTextures(1, &textureID);
glBindTexture(GL_TEXTURE_2D, textureID);

// Texture options, there are many, look it up
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);	
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);

// Load image data
int width, height, nrChannels;
unsigned char *data = stbi_load("texture.jpg", &width, &height, &nrChannels, 0);
if (data) {
  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, width, height, 0, GL_RGB, GL_UNSIGNED_BYTE, data);
  glGenerateMipmap(GL_TEXTURE_2D);
} else {
  std::cout << "Failed to load texture" << std::endl;
}
stbi_image_free(data);

// Later in the render part
glActiveTexture(GL_TEXTURE0);
glBindTexture(GL_TEXTURE_2D, textureID);
```
