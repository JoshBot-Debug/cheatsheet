### Context State Configuration
```cpp
glEnable(GL_DEPTH_TEST);  // Enable depth testing (prevents distant fragments overlapping close ones)
```

### Buffer Lifecycles (Generate $\rightarrow$ Bind $\rightarrow$ Allocate $\rightarrow$ Delete)
```cpp
GLuint VBO;
glGenBuffers(1, &VBO);
glBindBuffer(GL_ARRAY_BUFFER, VBO);
glBufferData(GL_ARRAY_BUFFER, sizeof(data), data, GL_STATIC_DRAW); // GL_STATIC_DRAW, GL_DYNAMIC_DRAW, GL_STREAM_DRAW
glDeleteBuffers(1, &VBO);
```

::::info
`GL_STATIC_DRAW`: I won't change the buffer data

`GL_DYNAMIC_DRAW`: I may change the buffer data it once in a while

`GL_STREAM_DRAW`: I will change the buffer data it every frame
::::

## Boilerplate

### Compiling a shader program
```cpp
#include <iostream>
#include <string>
#include <vector>
#include <GL/glew.h>

/**
 * @brief Compiles a single shader stage from source code string.
 * @param shaderType GL_VERTEX_SHADER | GL_FRAGMENT_SHADER | GL_GEOMETRY_SHADER | GL_TESS_CONTROL_SHADER | GL_TESS_EVALUATION_SHADER | GL_COMPUTE_SHADER
 * @param shaderSourceCode The raw GLSL source code string.
 * @return GLuint The compiled shader ID, or 0 if compilation failed.
 */
GLuint compileShaderStage(GLenum shaderType, const std::string& shaderSourceCode) {
  std::string stageName = "UNKNOWN";
  switch (shaderType) {
    case GL_VERTEX_SHADER:          stageName = "VERTEX"; break;
    case GL_TESS_CONTROL_SHADER:    stageName = "TESS_CONTROL"; break;
    case GL_TESS_EVALUATION_SHADER: stageName = "TESS_EVALUATION"; break;
    case GL_GEOMETRY_SHADER:        stageName = "GEOMETRY"; break;
    case GL_FRAGMENT_SHADER:        stageName = "FRAGMENT"; break;
    case GL_COMPUTE_SHADER:         stageName = "COMPUTE"; break;
  }

  // Create and source the shader
  GLuint shader = glCreateShader(shaderType);
  const char* sourceCStr = shaderSourceCode.c_str();
  glShaderSource(shader, 1, &sourceCStr, nullptr);
  
  // Compile
  glCompileShader(shader);

  // Check for compilation errors
  GLint success;
  glGetShaderiv(shader, GL_COMPILE_STATUS, &success);
  if (!success) {
    GLint logLength = 0;
    glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &logLength);
    
    std::vector<char> infoLog(logLength);
    glGetShaderInfoLog(shader, logLength, nullptr, infoLog.data());
    
    std::cerr << "ERROR::SHADER::" << stageName << "::COMPILATION_FAILED\n" 
              << infoLog.data() << "\n-----------------------------------------------------\n" 
              << std::endl;
              
    glDeleteShader(shader);
    return 0;
  }

  return shader;
}

/**
 * @brief Links multiple shader stages into a unified shader program.
 * @param shaderStages A vector containing compiled shader handles
 * @return GLuint The linked shader program ID, or 0 if linking failed.
 */
GLuint linkShaderProgram(const std::vector<GLuint>& shaderStages) {
  GLuint program = glCreateProgram();
  
  for (GLuint stage : shaderStages)
    if (stage != 0)
      glAttachShader(program, stage);
  
  glLinkProgram(program);
  
  GLint success;
  glGetProgramiv(program, GL_LINK_STATUS, &success);
  if (!success) {
    GLint logLength = 0;
    glGetProgramiv(program, GL_INFO_LOG_LENGTH, &logLength);
    
    if (logLength > 0) {
        std::vector<char> infoLog(logLength);
        glGetProgramInfoLog(program, logLength, nullptr, infoLog.data());
        std::cerr << "ERROR::SHADER::PROGRAM::LINKING_FAILED\n" 
                  << infoLog.data() << "\n-----------------------------------------------------\n" 
                  << std::endl;
    } else {
        std::cerr << "ERROR::SHADER::PROGRAM::LINKING_FAILED (Unknown Error)\n" << std::endl;
    }
    
    glDeleteProgram(program);
    return 0;
  }
  
  return program;
}

int main() {
  GLuint vs  = compileShaderStage(GL_VERTEX_SHADER,          vsSource);
  GLuint tcs = compileShaderStage(GL_TESS_CONTROL_SHADER,    tcsSource);
  GLuint tes = compileShaderStage(GL_TESS_EVALUATION_SHADER, tesSource);
  GLuint gs  = compileShaderStage(GL_GEOMETRY_SHADER,        gsSource);
  GLuint fs  = compileShaderStage(GL_FRAGMENT_SHADER,        fsSource);
  GLuint cs  = compileShaderStage(GL_COMPUTE_SHADER,         csSource);

  GLuint program = linkShaderProgram({ vs, tcs, tes, gs, fs, cs }); // FYI, this is just an example, cs shaders are generally in a seperate pipeline.

  // Gotta free memory, don't forget
  glDeleteShader(vs);
  glDeleteShader(tcs);
  glDeleteShader(tes);
  glDeleteShader(gs);
  glDeleteShader(fs);
  glDeleteShader(cs);

  // eventually
  glUseProgram(program);
}
```

### Allocating and Binding Mesh Data (`VAO`/`VBO`/`EBO`)

```cpp
std::vector<GLfloat> vertices = {0.5f,  0.5f, 0.0f, 0.5f, -0.5f, 0.0f, -0.5f, -0.5f, 0.0f, -0.5f,  0.5f, 0.0f};
std::vector<GLuint> indices = {  0, 1, 3, 1, 2, 3};

GLuint VAO, VBO, EBO;
glGenVertexArrays(1, &VAO);
glGenBuffers(1, &VBO);
glGenBuffers(1, &EBO);

glBindVertexArray(VAO);

glBindBuffer(GL_ARRAY_BUFFER, VBO);
glBufferData(GL_ARRAY_BUFFER, vertices.size() * sizeof(GLfloat), vertices.data(), GL_STATIC_DRAW);

glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(GLuint), indices.data(), GL_STATIC_DRAW);


// NOTE: Stride is 5 * sizeof(GLfloat) | [x,y,x,u,v]
// Location = 0
// First 3 floats are x,y,z
glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 5 * sizeof(GLfloat), (void*)0);
glEnableVertexAttribArray(0);

// Location = 1
// Following 2 floats are texture coords (UV)
glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 5 * sizeof(GLfloat), (void*)(3 * sizeof(GLfloat)));
glEnableVertexAttribArray(1);

// Unbind VAO & VBO
// DO NOT UNBIND THE EBO, nothing will render :P
// You can unbind the EBO after ending the recording though. But best practice is to never unbind the EBO, doing that is redundant, simply modifies OpenGL state.
glBindVertexArray(0);
glBindBuffer(GL_ARRAY_BUFFER, 0); 
```

::::info
**Multiple `VBO` one `VAO`** & **Multiple `EBO` one `VAO`**

**Multiple `VBO` one `VAO`**:
  * You can use multiple `VBOs` while recording a `VAO`. This is actually a nice practice.
  * *Location = 0* `VBO_postions` for vertex data
  * *Location = 1* `VBO_textures` for texture info
  * *Location = 2* `VBO_normals` for normals

**Multiple `EBO` one `VAO`**:
  * This is **NOT POSSIBLE**
  * You can only bind one `EBO` while recording a `VAO`

**Example**:
```cpp
// Separate raw data streams
std::vector<GLfloat> positions = {};
std::vector<GLfloat> texCoords = {};
std::vector<GLfloat> normals = {};
std::vector<GLuint> indices = {};

// Allocate the Object Handles
GLuint VAO;
GLuint VBO_Positions, VBO_TexCoords, VBO_Normals;
GLuint EBO;

glGenVertexArrays(1, &VAO);
glGenBuffers(1, &VBO_Positions);
glGenBuffers(1, &VBO_TexCoords);
glGenBuffers(1, &VBO_Normals);
glGenBuffers(1, &EBO);

// Start recording
glBindVertexArray(VAO);

// [VBO 1] Setup Positions & link to layout location = 0
glBindBuffer(GL_ARRAY_BUFFER, VBO_Positions);
glBufferData(GL_ARRAY_BUFFER, positions.size() * sizeof(GLfloat), positions.data(), GL_STATIC_DRAW);
glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(GLfloat), (void*)0);
glEnableVertexAttribArray(0);

// [VBO 2] Setup Textures & link to layout location = 1
glBindBuffer(GL_ARRAY_BUFFER, VBO_TexCoords);
glBufferData(GL_ARRAY_BUFFER, texCoords.size() * sizeof(GLfloat), texCoords.data(), GL_STATIC_DRAW);
glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 2 * sizeof(GLfloat), (void*)0); // Size is 2 for U,V!
glEnableVertexAttribArray(1);

// [VBO 3] Setup Normals & link to layout location = 2
glBindBuffer(GL_ARRAY_BUFFER, VBO_Normals);
glBufferData(GL_ARRAY_BUFFER, normals.size() * sizeof(GLfloat), normals.data(), GL_STATIC_DRAW);
glVertexAttribPointer(2, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(GLfloat), (void*)0);
glEnableVertexAttribArray(2);

// [EBO] Single shared index buffer
glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(GLuint), indices.data(), GL_STATIC_DRAW);

// Stop recording
glBindVertexArray(0);
glBindBuffer(GL_ARRAY_BUFFER, 0);
```
::::

### Texturing Pipeline

```cpp
#include <stb_image.h>

GLuint textureID;
glGenTextures(1, &textureID);
glBindTexture(GL_TEXTURE_2D, textureID);

// Texture options, there are many, look it up
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);	
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);

// Load image data
int width, height, nrChannels;
unsigned char *data = stbi_load("texture.jpg", &width, &height, &nrChannels, 0);
if (data) {
  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, width, height, 0, GL_RGB, GL_UNSIGNED_BYTE, data);
  glGenerateMipmap(GL_TEXTURE_2D);
} else {
  std::cout << "Failed to load texture" << std::endl;
}
stbi_image_free(data);

// Later in the render part
glActiveTexture(GL_TEXTURE0);
glBindTexture(GL_TEXTURE_2D, textureID);
```
